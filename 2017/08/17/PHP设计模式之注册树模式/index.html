注册树模式

单例模式解决的是如何在整个项目中创建唯一对象实例的问题，工厂模式解决的是如何不通过 new 建立实例对象的方法。而注册树模式考虑的是如何统筹管理已经实例化后的对象。

不管你是通过单例模式还是工厂模式还是二者结合生成的对象，首先都放在注册树上。当我向要使用某个对象时，直接从注册树上取一下就好。这和我们使用全局变量一样的方便实用。（注册的含义和观察者模式中注册观察者类似）

组成

需要一个注册树类，类内部要有一个数组存放注册的对象，还要有 set()，get()，_unset()函数用于实现注册，获取和销毁功能（这一点又有点像是 key-value 的数据库）。

实例1–单独使用注册树模式

<?php
// 注册树类
class Register{
    // 用于存储实例
    protected static $objects;

    // 将实例插入注册树中，用$alias作为别名
    public static function set($alias, $object) {
        self::$objects[$alias] = $object;
    }

    // 通过别名，从注册树中读取实例
    public static function get($alias) {
        if (isset(self::$objects[$alias])) {
            return self::$objects[$alias];
        } else {
            echo '您要找的对象实例不在注册树上<br>';
        }
    }

    // 销毁注册树中的实例
    public static function _unset($alias) {
        unset(self::$objects[$alias]);
    }
}

// 演示类
class demo {
    // 测试方法
    public function test() {
        echo 'test<br>';
    }
}

// 实例化测试类，获取对象实例
$demo = new demo();
// 注册到树上
$tree = Register::set('de', $demo);
// 取出来
$de = Register::get('de');
// 测试
$de->test();
// 销毁
Register::_unset('de');
// 尝试再次取出来
$de_two = Register::get('de');
// 尝试再次测试
$de_two->test();

执行结果如下：

// 从注册树上取下并调用test函数
test
// 销毁后调用get方法，无法取到
您要找的对象实例不在注册树上
// 因为没有对象，所以无法调用对象方法，报错
Fatal error: Call to a member function test() on a non-object in test.php on line 77

实例2–配合单例模式和工厂模式共同使用

注册数类和之前一样，不需要改动

<?php
// 注册树类
class Register{
    // 用于存储实例
    protected static $objects;

    // 将实例插入注册树中，用$alias作为别名
    public static function set($alias, $object) {
        self::$objects[$alias] = $object;
    }

    // 通过别名，从注册树中读取实例

            return self::$objects[$alias];

        }
    }

    // 销毁注册树中的实例

        unset(self::$objects[$alias]);
    }
}

通过单例模式实现的 MySQL 类作为演示对象

class MySQL {
    private $_DBConfig;
    private static $_instance = null;

    private function __construct($config) {
        $this->_DBConfig = serialize($config);
    }

    private function __clone() {}

    // 单例模式
    public static function getInstance($config) {
        if (!(self::$_instance instanceof self)) {
            self::$_instance = new self($config);
        }
        return self::$_instance;
    }

    public function getDBConfig() {
        return $this->_DBConfig;
    }
}

通过工厂模式实现 MySQL 对象的实例化

class Factory {
    public static function create($config) {
        return MySQL::getInstance($config);
    }
}

客户端代码，先通过工厂方法实现实例化后注册到注册树上，然后取出实例执行演示方法。

class Client {
    public static function main() {
        $config = array("localhost", "3306", "root", " ");
        // 加入注册树
        Register::set('Mysql', Factory::create($config));
        // 取出对象实例
        $object = Register::get('Mysql');
        // 执行测试方法
        echo $object->getDBConfig();
    }
}

Client::main();

最后输出数据库配置信息：

a:4:{i:0;s:9:"localhost";i:1;s:4:"3306";i:2;s:4:"root";i:3;s:1:" ";}

也就是我们的 config 数组的序列化字符串。
